BuiToggle

HTML

	<script type="text/javascript" defer src="/assets/js/module.bui.js"></script>
<script type="text/javascript" defer src="/assets/js/module.bui.template.js"></script>

<div id="targetElementId" data-bui-toggle="toggle-selector" data-bui-toggle-target="targetElementId">toggle target</div>
<a href="#targetElementId" data-bui-toggle-button="targetElementId">toggle button</a>

javascript

default

var toggleName = new buiToggle('[data-bui-toggle="toggle-selector"]');

options

var toggleName = new buiToggle('[data-bui-toggle="toggle-selector"]', {
// general
mode: 'normal',
event: 'click',
active: false,
activeClass: 'active',

disabled: false,
disabledClass: null,

focusin: false,
focusout: false,

clickout: false,
clickoutTarget: null,

// target
target: null,
targetClass: 'bui-toggle-target',
targetActiveClass: 'active',
targetAttribute: 'data-toggle-target',

// close
close: false,
closeButton: 'button',
closeButtonArea: null,
closeButtonClass: 'close',
closeButtonText: 'close',

// react target
reactTarget: null,
reactTargetClass: null,
reactTargetActiveClass: 'active',

// react Parent
reactParent: null,
reactParentClass: null,
reactParentActiveClass: 'active',

tooltipContainer: '#page',
tooltipArrowClass: 'tooltip-arrow',

/* callback */
onloadCallBack: function() {return false;},
eventCallBack: function() {return false;},
activeCallBack: function() {return false;},
inactiveCallBack: function() {return false;}
});

API

// toggle actions
toggleName.toggle('selector');

// add to toggle plugins
toggleName.appendToggle('selector');

BuiTab

HTML

	<script type="text/javascript" defer src="/assets/js/module.bui.js"></script>
	<script type="text/javascript" defer src="/assets/js/module.bui.template.js"></script>
	
	<div class="tab-display" data-bui-tab="normal">
	<ul class="tab-list">
	<li class="tab-item"><a class="tab-name" href="#tabTarget1"></a></li>
	<li class="tab-item"><a class="tab-name" href="#tabTarget2"></a></li>
	<li class="tab-item"><a class="tab-name" href="#tabTarget3"></a></li>
	</ul>
	</div>
	
	<div class="tabTarget1">Tab Target 1</div>
	<div class="tabTarget2">Tab Target 2</div>
	<div class="tabTarget3">Tab Target 3</div>

javascript

default

var tabs = new buiTab('[data-bui-tab="normal"]');

options

var tabs = new buiTab('[data-bui-tab="normal"]'), {
// general
active: false,
activeClass: 'current',
initial: 0,

// tab
tabItem: '.tab-item',
tabName: '.tab-name',

// target
target: null,
targetClass: 'bui-tab-target',
targetActiveClass: 'active',
});
